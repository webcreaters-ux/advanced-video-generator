!pip install advanced-video-generator

# Import and use
from advanced_video_generator import create_video_generator

Command Line Usage 
# Install
pip install advanced-video-generator

# Generate a video
python run.py --script my_script.txt --output video.mp4

# Start web UI
python run.py --web-ui

# Batch processing
python run.py --batch --script-dir ./scripts
Python API 
from advanced_video_generator import (
    create_video_generator,
    GenerationOptions,
    VideoQuality
)

# Create generator
generator = create_video_generator()

# Configure options
options = GenerationOptions(
    quality=VideoQuality.HIGH,
    generate_images=True,
    add_subtitles=True,
    voice_engine="google"
)

# Generate video
result = generator.generate_from_script(
    script_text="Your script text...",
    output_path="output.mp4",
    options=options
)

print(f"Video saved to: {result['output_path']}")
